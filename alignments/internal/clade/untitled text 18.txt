 #~# Root (Family)
  
  # Export the recursively populated root gene alignments 
  module fastaAlignmentExporter
 	export AL_Parvoviridae_MASTER -r REF_Proto_MASTER_CPV -f whole_genome --recursive -a  -e -o alignments/tree/Parvoviridae-all.aln.fna
	exit

  # Re-import the recursively populated root alignment into the project
  module parvoTreeAlignmentImporterNucs import AL_UNC_TREE_Parvoviridae -f alignments/tree/Parvoviridae-all.aln.fna
  create alignment AL_TREE_Parvoviridae -r REF_Proto_MASTER_CPV
  
  alignment AL_TREE_Parvoviridae
	derive segments AL_UNC_TREE_Parvoviridae -a --mergeStrategy OVERWRITE
	exit
	
  module parvoFeaturePresenceRecorder
    record feature-presence AL_TREE_Parvoviridae --featureName whole_genome --descendentFeatures
    exit

  
  # Create the phylogenies
  module raxmlPhylogenyGenerator

    generate nucleotide phylogeny AL_TREE_Parvoviridae -s alignmentColumnsSelectorParvoviridaeRep78 \
      -w "fLocNotes.featureLoc.feature.name = 'Rep78' and fLocNotes.ref_nt_coverage_pct >= 10" \
      -o trees/family-level/Parvoviridae-Rep78.export_nucs.tre NEWICK_BOOTSTRAPS

    exit
